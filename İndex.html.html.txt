    <script>
        // 1. YouTube API Yükleme
        var tag = document.createElement('script');
        tag.src = "https://www.youtube.com/iframe_api";
        var firstScriptTag = document.getElementsByTagName('script')[0];
        firstScriptTag.parentNode.insertBefore(tag, firstScriptTag);

        var player;
        function onYouTubeIframeAPIReady() {
            player = new YT.Player('youtube-player', {
                height: '1',
                width: '1',
                videoId: 'gRjsXfD6ivo', // Özdemir Erdoğan - Bana Ellerini Ver
                playerVars: {
                    'autoplay': 0,
                    'controls': 0,
                    'loop': 1,
                    'playlist': 'gRjsXfD6ivo' // Döngü için gerekli
                }
            });
        }

        // 2. Butona Tıklanınca Intro Kapanır ve Müzik Başlar
        document.getElementById('enter-btn').addEventListener('click', function() {
            // Müzik Başlat
            if (player && player.playVideo) {
                player.playVideo();
                player.unMute(); // Sesi açtığından emin ol
                player.setVolume(100);
            }

            // Ekran Geçişi
            document.getElementById('intro-screen').style.transform = 'translateY(-100%)';
            document.getElementById('main-content').style.opacity = '1';
            
            setInterval(createParticle, 200);
        });

        // 3. Zaman Hesaplama (Aynı Mantık)
        const startDate = new Date("2025-10-31T21:00:00");
        function updateTimer() {
            const now = new Date();
            let years = now.getFullYear() - startDate.getFullYear();
            let months = now.getMonth() - startDate.getMonth();
            let days = now.getDate() - startDate.getDate();
            if (days < 0) { months--; let pMonth = new Date(now.getFullYear(), now.getMonth(), 0); days += pMonth.getDate(); }
            if (months < 0) { years--; months += 12; }
            const diffMs = now - startDate;
            const hours = Math.floor((diffMs / (1000 * 60 * 60)) % 24);
            const minutes = Math.floor((diffMs / (1000 * 60)) % 60);
            const seconds = Math.floor((diffMs / 1000) % 60);
            document.getElementById("years").innerText = Math.max(0, years);
            document.getElementById("months").innerText = Math.max(0, months);
            document.getElementById("days").innerText = Math.max(0, days);
            document.getElementById("hours").innerText = Math.max(0, hours);
            document.getElementById("minutes").innerText = Math.max(0, minutes);
            document.getElementById("seconds").innerText = Math.max(0, seconds);
        }
        setInterval(updateTimer, 1000);
        updateTimer();

        // 4. Efektler
        function createParticle() {
            const p = document.createElement("div");
            p.classList.add("particle");
            p.innerHTML = Math.random() > 0.5 ? "❤️" : "✨";
            p.style.left = Math.random() * 100 + "vw";
            p.style.fontSize = (Math.random() * 20 + 10) + "px";
            p.style.animationDuration = (Math.random() * 3 + 4) + "s";
            document.body.appendChild(p);
            setTimeout(() => p.remove(), 6000);
        }
